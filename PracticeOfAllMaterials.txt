Exercise 1:
  1) Create a table called P1E1_Products with the following columns:
    |ProductID (integer, primary key)
    |ProductName (varchar2)
    |Category (varchar2)
    |Price (varchar2)
  2) Add a new column Country (varchar2) to the P1E1_Products table
  3) Insert one product into the P1E1_Products table:
    |ProductID: 1, ProductName: 'Laptop', Category: 'Electronics', Price: 899.99, Country: 'USA'
  4) Insert four more products into the P1E1_Products table:
    |ProductID: 2, ProductName: 'Book', Category: 'Books', Price: 19.99, Country: 'USA' 
    |ProductID: 3, ProductName: 'T-shirt', Category: 'Apparel', Price: 14.99, Country: 'Turkey' 
    |ProductID: 4, ProductName: TV, Category: Electronics, Price: 699.99, Country: 'Japan' 
    |ProductID: 5, ProductName: 'Dress', Category: 'Apparel', Price:99.99, Country: 'China'
  5) Update the price of the 'Laptop' to 999.99 in the P1E1_Products table
  6) Retrieve all products in the 'Electronics' category from the P1E1_Products table.
  7) Delete the product with the price < 20from the P1E1_Products table
  8) Remove all rows from the P1E1_Products table
  9) Delete P1E1_Products table

Result of Exercise 1:
  1)  |Create table P1E1_Products(
      |ProductId int Primary key,
      |ProductName Varchar2(50),
      |Category Varchar2(50),
      |Price Varchar2(50)
      |);

  2)  |Alter table P1E1_Products add Country Varchar2(50);

  3)  |INSERT INTO P1E1_Products(ProductId, ProductName, Category, Price, Country) VALUES (1, 'Laptop', 'Electronics', 899.99, 'USA'),

  4)  |INSERT ALL
      |INTO P1E1_Products(ProductId, ProductName, Category, Price, Country) VALUES (2, 'Book', 'Books', 19.99, 'USA')
      |INTO P1E1_Products(ProductId, ProductName, Category, Price, Country) VALUES (3, 'T-shirt', 'Apparel', 14.99, 'Turkey')
      |INTO P1E1_Products(ProductId, ProductName, Category, Price, Country) VALUES (4, 'TV', 'Electronics', 699.99, 'Japan')
      |INTO P1E1_Products(ProductId, ProductName, Category, Price, Country) VALUES (5, 'Dress', 'Apparel', 99.99, 'China')
      |Select * from P1E1_Products;

  5)  |Update P1E1_Products 
      |set Price = 999.99
      |where ProductName = 'Laptop';

Exercise 2:
  1) Create a table called P1E2_Movies with the following columns:
    |MovieID (integer, primary key)
    |Title (varchar2)
    |Director (varchar2)
    |Genre (varchar2)
    |ReleaseYear (integer)
  2) Add a new column Rating(decimal) to the P1E2_Movies table
  3) Insert six movie records into the P1E2_Movies table:
    |MovieID: 1, Title: 'Inception', Director: 'Christopher Nolan', Genre: 'Science Fiction', ReleaseYear: 2010, Rating: 8.8
    |MovieID: 2, Title: 'The Shawshank Redemption', Director: 'Frank Darabont', Genre: 'Drama', ReleaseYear: 1994, Rating: 9.3 
    |MovieID: 3, Title: 'The Dark Knight', Director: 'Christopher Nolan', Genre: 'Action', ReleaseYear: 2008, Rating: 9.0
    |MovieID: 4, Title: 'Pulp Fiction', Director: 'Quentin Tarantino', Genre: 'Crime', ReleaseYear: 1994, Rating: 8.9
		|MovieID: 5, Title: 'Forrest Gump', Director: 'Robert Zemeckis', Genre: 'Drama', ReleaseYear: 1994, Rating: 8.8
    |MovieID: 6, Title: 'The Matrix', Director: 'Lana Wachowski, Lilly Wachowski', Genre: 'Science Fiction', ReleaseYear: 1999, Rating: 8.7
  4) Update the Rating of the 'Inception' to 9.0 in the P1E2_Movies table
  5) Retrieve all movies with a Rating greater than or equal to 9.0, ordered by ReleaseYear in descending order.
  6) Delete the movie with MovieID 4 from the P1E2_Movies table.
  7) Retrieve the average rating of all movies in the 'Drama' genre.
  8) Remove all rows from the P1E2_Movies table
  9) Delete the P1E2_Movies table


Result of Exercise 2:
  1)  |Create table P1E2_Movies(
      |MovieID int primary key,
      |Title varchar2(50),
      |Director varchar2(50),
      |Genre varchar2(50),
      |ReleaseYear int
      |);

  2)  |Alter table P1E2_Movies add Rating decimal(10, 2);

  3)  |INSERT ALL
      |INTO P1E2_Movies(MovieID, Title, Director, Genre, ReleaseYear, Rating) VALUES (1, 'Inception', 'Christopher Nolan', 'Science Fiction', 2010, 8.8)
      |INTO P1E2_Movies(MovieID, Title, Director, Genre, ReleaseYear, Rating) VALUES (2, 'The Shawshank Redemption', 'Frank Darabont', 'Drama', 1994, 9.3)
      |INTO P1E2_Movies(MovieID, Title, Director, Genre, ReleaseYear, Rating) VALUES (3, 'The Dark Knight', 'Christopher Nolan', 'Action', 2008, 9.0)
      |INTO P1E2_Movies(MovieID, Title, Director, Genre, ReleaseYear, Rating) VALUES (4, 'Pulp Fiction', 'Quentin Tarantino', 'Crime', 1994, 8.9)
      |INTO P1E2_Movies(MovieID, Title, Director, Genre, ReleaseYear, Rating) VALUES (5, 'Forrest Gump', 'Robert Zemeckis', 'Drama', 1994, 8.8)
      |INTO P1E2_Movies(MovieID, Title, Director, Genre, ReleaseYear, Rating) VALUES (6, 'The Matrix', 'Lana Wachowski, Lilly Wachowski', 'Science Fiction', 1999, 8.7)
      |Select * from P1E2_Movies;

  4)  |Update P1E2_Movies
      |set Rating = 9.0
      |where Title = 'Inception';

  5)  |Select * from P1E2_Movies
      |where Rating >= 9.0 
	    |order by ReleaseYear DESC;

  6)  |Delete from P1E2_Movies where MovieID = 4;

	7)  |SELECT AVG(rating) AS average_rating
      |FROM P1E2_Movies
	    |WHERE genre = 'Drama';

  8)  |Truncate table P1E2_Movies;

  9)  |Drop table P1E2_Movies;
