<script>
    document.addEventListener("DOMContentLoaded", function () {
        const projectItems = document.querySelectorAll(".nav-item");
        const projectDetailsCanvas = new bootstrap.Offcanvas(document.getElementById('projectDetailsCanvas'));
    
        projectItems.forEach(item => {
            item.addEventListener("click", function () {
                // Remove selected-project class from all other items
                projectItems.forEach(otherItem => {
                    otherItem.classList.remove('selected-project');
                });
    
                // Add selected-project class to the clicked item
                this.classList.add('selected-project');
    
                const projectName = this.querySelector(".project-name").textContent;
                // Update selected project display
                document.getElementById("selectedProjectName").textContent = projectName;
                // Update project details offcanvas
                document.getElementById("projectDetailsLabel").textContent = projectName;

    
                // Show the project details offcanvas
                projectDetailsCanvas.show();
            });
        });
    });
    
    
</script>
